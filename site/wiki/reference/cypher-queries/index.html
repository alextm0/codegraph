    Cypher Query Reference

    Useful Cypher queries for exploring and inspecting the CodeGraph Neo4j database. Run these in the Neo4j Browser (http://localhost:7474) or via the Neo4j Python driver.

    Graph Overview

    // Count all nodes by type
MATCH (n)
RETURN labels(n) AS type, count(*) AS count
ORDER BY count DESC

// Count all edges by type
MATCH ()-[r]->()
RETURN type(r) AS relation, count(*) AS count
ORDER BY count DESC

// Total nodes and edges
MATCH (n) WITH count(n) AS nodes
MATCH ()-[r]->() WITH nodes, count(r) AS edges
RETURN nodes, edges

    Finding Entities

    // Find a function/method by name
MATCH (n {name: "authenticate"})
RETURN labels(n), n.name, n.file_path, n.line_number

// Find all methods of a class
MATCH (c:Class {name: "AuthService"})-[:CONTAINS]->(m:Method)
RETURN m.name, m.line_number, m.signature
ORDER BY m.line_number

// Find all classes in a file
MATCH (f:File {name: "auth_service.py"})-[:CONTAINS]->(c:Class)
RETURN c.name, c.line_number

// Find entities matching a partial name (case insensitive)
MATCH (n)
WHERE toLower(n.name) CONTAINS "auth"
RETURN labels(n), n.name, n.file_path
LIMIT 20

    Call Graph Navigation

    // What does this function call? (downstream dependencies)
MATCH (f {name: "authenticate"})-[:CALLS]->(callee)
RETURN callee.name, callee.file_path, callee.line_number

// What calls this function? (upstream — who depends on it)
MATCH (caller)-[:CALLS]->(f {name: "authenticate"})
RETURN caller.name, caller.file_path

// Full call chain — 2 hops downstream
MATCH path = (start {name: "authenticate"})-[:CALLS*1..2]->(end)
RETURN [n IN nodes(path) | n.name] AS chain

// Find the most-called functions (hotspots)
MATCH ()-[:CALLS]->(f)
RETURN f.name, f.file_path, count(*) AS call_count
ORDER BY call_count DESC
LIMIT 20

// Find functions that call nothing (leaf nodes)
MATCH (f:Function)
WHERE NOT (f)-[:CALLS]->()
RETURN f.name, f.file_path
LIMIT 20

// Detect potential circular dependencies (calls that form cycles)
MATCH path = (a)-[:CALLS*2..5]->(a)
RETURN [n IN nodes(path) | n.name] AS cycle
LIMIT 10

    Import Graph Navigation

    // What does this file import?
MATCH (f:File {name: "auth_service.py"})-[:IMPORTS]->(dep:File)
RETURN dep.name, dep.file_path

// What files import from this file?
MATCH (f:File)-[:IMPORTS]->(target:File {name: "user.py"})
RETURN f.name, f.file_path

// Find the most-imported files (high fan-in = central modules)
MATCH ()-[:IMPORTS]->(f:File)
RETURN f.name, count(*) AS import_count
ORDER BY import_count DESC
LIMIT 10

// Find files that import nothing from the project (entry points or utilities)
MATCH (f:File)
WHERE NOT (f)-[:IMPORTS]->()
RETURN f.name, f.file_path

// Find the deepest import chains
MATCH path = (a:File)-[:IMPORTS*]->(z:File)
WHERE NOT (z)-[:IMPORTS]->()
RETURN length(path) AS depth, [n IN nodes(path) | n.name] AS chain
ORDER BY depth DESC
LIMIT 5

    Inheritance Graph

    // Find all classes that inherit from a base class
MATCH (child:Class)-[:INHERITS_FROM]->(parent:Class {name: "BaseService"})
RETURN child.name, child.file_path

// Find the full inheritance hierarchy (multi-level)
MATCH path = (leaf:Class)-[:INHERITS_FROM*]->(root:Class)
WHERE NOT (root)-[:INHERITS_FROM]->()
RETURN [n IN nodes(path) | n.name] AS hierarchy

// Find classes with multiple inheritance
MATCH (c:Class)-[:INHERITS_FROM]->(p:Class)
WITH c, count(p) AS parent_count
WHERE parent_count > 1
RETURN c.name, c.file_path, parent_count

    Graph Health & Diagnostics

    // Check for nodes with no edges (orphans — parser may have failed)
MATCH (n)
WHERE NOT (n)-[]-() AND NOT ()-[]-(n)
RETURN labels(n), n.name, n.file_path
LIMIT 20

// Find files with very high connectivity (god files)
MATCH (f:File)-[r]-()
RETURN f.name, count(r) AS connections
ORDER BY connections DESC
LIMIT 10

// Check edge weight distribution
MATCH ()-[r]->()
RETURN type(r), min(r.weight), max(r.weight), avg(r.weight)

// Count nodes by file (which files have the most entities?)
MATCH (f:File)-[:CONTAINS]->(n)
RETURN f.name, count(n) AS entity_count
ORDER BY entity_count DESC
LIMIT 15

    GDS Projection Management

    // Check if the projection exists
CALL gds.graph.exists('codegraph')
YIELD exists

// List all projections
CALL gds.graph.list()
YIELD graphName, nodeCount, relationshipCount, creationTime

// Drop and recreate the projection (needed after graph updates)
CALL gds.graph.drop('codegraph', false)
YIELD graphName

// Create the projection used by PPR
CALL gds.graph.project(
  'codegraph',
  ['Function', 'Method', 'Class', 'File'],
  {
    CALLS:         {orientation: 'UNDIRECTED', properties: 'weight'},
    IMPORTS:       {orientation: 'UNDIRECTED', properties: 'weight'},
    INHERITS_FROM: {orientation: 'UNDIRECTED', properties: 'weight'},
    CONTAINS:      {orientation: 'UNDIRECTED', properties: 'weight'}
  }
)

    Running PPR Manually

    // Run PPR from a specific seed
MATCH (seed:Method {name: 'authenticate'})
CALL gds.pageRank.stream('codegraph', {
  maxIterations: 20,
  dampingFactor: 0.85,
  sourceNodes: [seed],
  relationshipWeightProperty: 'weight'
})
YIELD nodeId, score
RETURN gds.util.asNode(nodeId).name AS name,
       gds.util.asNode(nodeId).file_path AS file,
       score
ORDER BY score DESC
LIMIT 20

    Maintenance

    // Clear all data (destructive — use with caution)
MATCH (n) DETACH DELETE n

// Verify GDS version
RETURN gds.version()

// Check database info
CALL dbms.components()
YIELD name, versions
